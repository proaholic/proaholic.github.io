function getHashTable($str,$length=200)
{
    $crc = abs(crc32($str));
    if( $crc & 0x80000000){
        $crc ^= 0xffffffff;
        $crc += 1;
    }
    $hash = $crc%$length;
    $hash = sprintf('%02d',$hash);
    return $hash;
}
